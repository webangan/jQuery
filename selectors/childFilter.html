:first-child Selector
$(":first-child").css("background","#f00"); 		Selects all elements that are the first child of their parent.
:last-child Selector
$(":last-child").css("background","#f00") 		Selects all elements that are the last child of their parent.
:nth-child() Selector
$(":nth-child(3)").css("background","#f00");
jQuery( ":nth-child(index/even/odd/equation)" ) 		Selects all elements that are the nth-child of their parent start at 1
:nth-last-child() Selector
$(":nth-last-child(3)").css("background","#f00");
jQuery( ":nth-last-child(index/even/odd/equation)" ) 		Selects all elements that are the nth-child of their parent, counting from the last element to the first.
:first-of-type Selector
$("div:first-of-type").css("background","#f00"); 		Selects all elements that are the first among siblings of the same element name.
:last-of-type Selector
$("div:last-of-type").css("background","#f00"); 		Selects all elements that are the last among siblings of the same element name.
nth-of-type() Selector
$(":nth-of-type(2)").css("background","#f00");
jQuery( ":nth-of-type(index/even/odd/equation)" ) 		Selects all elements that are the nth child of their parent in relation to siblings with the same element name.
nth-last-of-type() Selector
jQuery( ":nth-last-of-type(index/even/odd/equation)" )
Selects all the elements that are the nth-child of their parent in relation to siblings with the same element name, counting from the last element to the first.
:only-child Selector
$(":only-child").css("background","#f00"); 		Selects all elements that are the only child of their parent.
:first-of-type Selector
$("p:only-of-type").css("background","#f00"); 		Selects all elements that have no siblings with the same element name.
